function moveField(parentField, oldIndex, newIndex) {
    var instance = this;

    var fields = parentField.get('fields');

    fields.splice(newIndex, 0, fields.splice(oldIndex, 1)[0]);

    instance.recreateEditors(fields[newIndex]);
}
