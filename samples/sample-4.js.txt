const users = [
    {id: 'ad1bf863', name: 'Jane', role: 'admin'},
    {id: '6cf662a1', name: 'John', role: 'admin'},
    {id: '371cc472', name: 'Mary', role: 'moderator'},
    // ...
];

function getUser(uuid) {
    return users.find(({id}) => id === uuid);
}
